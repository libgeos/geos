418 template <class T>
419 std::unique_ptr<Geometry>
420 buildGeometry(T from, T toofar) const
421 {
422 bool isHeterogeneous = false;
423 std::size_t count = 0;
424 int geomClass = -1;
425 for(T i = from; i != toofar; ++i) {
426 ++count;
427 const Geometry* g = *i;
428 if(geomClass < 0) {
429 geomClass = g->getSortIndex();
430 }
431 else if(geomClass != g->getSortIndex()) {
432 isHeterogeneous = true;
433 }
434 }
435
436 // for the empty geometry, return an empty GeometryCollection
437 if(count == 0) {
438 return std::unique_ptr<Geometry>(createGeometryCollection());
439 }
440
441 // for the single geometry, return a clone
442 if(count == 1) {
443 return (*from)->clone();
444 }
445
446 // Now we know it is a collection
447
448 // FIXME:
449 // Until we tweak all the createMulti* interfaces
450 // to support taking iterators we'll have to build
451 // a custom vector here.
452 std::vector<std::unique_ptr<Geometry>> fromGeoms;
453 for(T i = from; i != toofar; ++i) {
454 fromGeoms.push_back((*i)->clone());
455 }
456
457 // for an heterogeneous ...
458 if(isHeterogeneous) {
459 return createGeometryCollection(std::move(fromGeoms));
460 }
461
462 // At this point we know the collection is not hetereogenous.
463 switch((*from)->getDimension()) {
464 case Dimension::A: return createMultiPolygon(std::move(fromGeoms));
465 case Dimension::L: return createMultiLineString(std::move(fromGeoms));
466 case Dimension::P: return createMultiPoint(std::move(fromGeoms));
467 default:
468 throw geos::util::IllegalArgumentException(std::string("Invalid geometry type."));
469 }
470 }
