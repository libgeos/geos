930 template<typename T>
931 static int compare(const T& a, const T& b)
932 {
933 std::size_t i = 0;
934 std::size_t j = 0;
935 while(i < a.size() && j < b.size()) {
936 const auto& aGeom = *a[i];
937 const auto& bGeom = *b[j];
938
939 int comparison = aGeom.compareTo(&bGeom);
940 if(comparison != 0) {
941 return comparison;
942 }
943
944 i++;
945 j++;
946 }
947
948 if(i < a.size()) {
949 return 1;
950 }
951
952 if(j < b.size()) {
953 return -1;
954 }
955
956 return 0;
957 }
